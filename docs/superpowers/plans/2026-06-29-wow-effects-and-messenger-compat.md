# Wow-Effects, Interactivity & file:// Compatibility Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add scroll/hover/parallax visual polish and interactive elements (nav, back-to-top, animated counters, form-success animation) to the existing static prototype, and fix file:// compatibility so the page works when opened directly (e.g. received via messenger) instead of only through a local server.

**Architecture:** Pure CSS + vanilla JS, no new CDN dependencies. The browser-facing script switches from an ES module (`<script type="module">`, blocked under `file://`) to a generated classic script `js/bundle.js`, produced by a small shell script that concatenates the existing tested ES-module source files and strips `import`/`export`. New visual effects are added to `js/main.js` (the DOM-glue source file) and a new `css/effects.css`.

**Tech Stack:** Plain HTML5/CSS3/vanilla JS, Node's built-in test runner for the three pure utilities, a `bash` concatenation script (no bundler).

## Global Constraints

- No new CDN dependencies (Tailwind CDN was already removed for reliability — see `prototype/css/utilities.css`).
- Every new animation must respect `prefers-reduced-motion: reduce` — either by not running at all, or by snapping to the final visible state instantly (no permanently-hidden content if motion is disabled).
- Pointer-only effects only run under `(pointer: fine)`, consistent with the existing cursor-object/magnetic-button pattern in `js/main.js`.
- `prototype/js/magnetic.js`, `js/honeypot.js`, `js/kinetic-text.js`, and their tests in `prototype/tests/` are not modified — they stay the tested ES-module source of truth.
- `prototype/js/bundle.js` is generated, never hand-edited — `scripts/build-bundle.sh` regenerates it from the ES module sources.
- Existing section content/copy/structure (7 sections, pricing, FAQ) does not change — only new visual layers and the Hero counters row are added.
- Placeholder counter values (50/5/14) must be marked in a comment as values to replace with real data later.

**Files:**

**Interfaces:**

- Consumes: nothing new — reads the existing `prototype/js/*.js` files unchanged.

- [ ] **Step 1: Write the bundle script**

Create `scripts/build-bundle.sh`:

```bash
#!/usr/bin/env bash
# Concatenates the ES module source files in prototype/js/ into a single
# classic (non-module) script, prototype/js/bundle.js, so the page works
# when opened directly via file:// (browsers block `<script type="module">`
# under file://, but plain scripts are unaffected). Source files keep their
# export/import syntax untouched — Node still tests them as ES modules;
# this script only strips those two patterns in the generated copy.
set -euo pipefail

cd "$(dirname "$0")/../prototype"

{
  echo "// Generated by scripts/build-bundle.sh from js/kinetic-text.js, js/magnetic.js,"
  echo "// js/honeypot.js, js/main.js — do not edit by hand. Re-run the script after"
  echo "// changing any of those files."
  echo
  for file in js/kinetic-text.js js/magnetic.js js/honeypot.js js/main.js; do
    grep -v -E "^import .* from '.*';\$" "$file" | sed -E 's/^export //'
    echo
  done
} > js/bundle.js

echo "wrote prototype/js/bundle.js"
```

- [ ] **Step 3: Add the reveal observer to main.js**

```js
function setupScrollReveal() {
  const sections = document.querySelectorAll('.reveal');
  if (!sections.length || reduceMotion) return;

  sections.forEach((section) => section.classList.add('reveal-pending'));

  const observer = new IntersectionObserver(
    (entries) => {
      entries.forEach((entry) => {
        if (entry.isIntersecting) {
          entry.target.classList.remove('reveal-pending');
          entry.target.classList.add('reveal-visible');
          observer.unobserve(entry.target);
        }
      });
    },
    { threshold: 0.15 }
  );

  sections.forEach((section) => observer.observe(section));
}
```

function setupParallax() {
  const orbs = document.querySelectorAll('.glow-orb');
  if (!orbs.length || reduceMotion) return;

  let ticking = false;
  window.addEventListener('scroll', () => {
    if (ticking) return;
    ticking = true;
    requestAnimationFrame(() => {
      const offset = window.scrollY * 0.15;
      orbs.forEach((orb, index) => {
        const direction = index % 2 === 0 ? 1 : -1;
        orb.style.transform = `translateY(${offset * direction}px)`;
      });
      ticking = false;
    });
  });
}

function setupBackToTop() {

function animateCounter(element) {
  const target = Number(element.dataset.target);
  const suffix = element.dataset.suffix || '';
  const duration = 1200;
  const start = performance.now();

  function tick(now) {
    const progress = Math.min((now - start) / duration, 1);
    const value = Math.round(target * progress);
    element.textContent = `${value}${suffix}`;
    if (progress < 1) {
      requestAnimationFrame(tick);
    }
  }

  requestAnimationFrame(tick);
}

function setupCounters() {
  const counters = document.querySelectorAll('.counter');
  if (!counters.length) return;

  if (reduceMotion) {
    counters.forEach((counter) => {
      counter.textContent = `${counter.dataset.target}${counter.dataset.suffix || ''}`;
    });
    return;
  }

  const observer = new IntersectionObserver(
    (entries) => {
      entries.forEach((entry) => {
        if (entry.isIntersecting) {
          animateCounter(entry.target);
          observer.unobserve(entry.target);
        }
      });
    },
    { threshold: 0.5 }
  );

  counters.forEach((counter) => observer.observe(counter));
}
